51 
52  uint16_t max_len, chunk_len = LWIP_MIN(http_response_get_bytes_left(req->res), 0xffff);
53

55  return ERR_OK;
56  }
57 
58  max_len = altcp_sndbuf(pcb);
59  if (max_len < chunk_len) {

65  if (err == ERR_MEM) {

67  chunk_len = 1;
68  } else {
69  chunk_len /= 2;
70  }
71  }
72  } while ((err == ERR_MEM) && (chunk_len > 1));
73

75  req->res->data_ptr_index += chunk_len;
76  }
