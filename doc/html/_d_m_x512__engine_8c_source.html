219 DMX512_engine_err_e DMX512_engine_load_config(TCHAR *config_file_path){
220 
221  DMX512_engine_err_e err = DMX512_ENGINE_CONFIG_LOAD_EXCEPTION;
222  FIL config_file;
223 
224  if(fs_get_mount_status() != FS_MOUNTED){ fs_init(); }
225 
226  if(f_open(&config_file, config_file_path, FA_OPEN_ALWAYS | FA_READ ) != FR_OK){
227  }else if(_DMX512_engine_load_config_header(&config_file) != DMX512_ENGINE_OK){
228  }else if(_DMX512_engine_load_config_patch(&config_file) != DMX512_ENGINE_OK){
229  }else if(_DMX512_engine_load_config_scenes(&config_file) != DMX512_ENGINE_OK){
230  }else if(_DMX512_engine_load_config_chasers(&config_file) != DMX512_ENGINE_OK){
231  }else{
232  err = DMX512_ENGINE_OK;
233  f_close(&config_file);
234  }
235 
236  return err;
237 }
